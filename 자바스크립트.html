1. var,let,const 차이점

            var : 재선언, 재할당(값변경) 가능 (사용을 거의 하지 않음)
                ex) var 변수 - 0; 
                    var 변수 = 1; //재선언 가능
                        변수 = 2; //재할당 가능
            let : 재선언불가, 재할당(값변경) 가능 (java의 변수같은 존재)
                ex) let 변수 = 0;
                    let 변수 = 1; //재선언 불가
                        변수 = 2; //재할당 가능
            const : 재선언불가, 재할당(값변경) 불가 (java의 상수 같은 존재)
                ex) const 변수 = 0;
                    const 변수 = 1; //재선언 불가
                          변수 = 2; //재할당 불가
            
            재선언과 재할당이란?
            재선언 : 선언 = 자신의 생각이나 입장을 분명하게 하는 것
                    다시 변수명에 대해서 정의를 내리는 것
            재할당 : 값을 다시 새로운 값을 저장하는 것
자료형 재선언 재할당
